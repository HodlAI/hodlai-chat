 import requests
from eth_account.messages import encode_defunct
from web3 import Web3
import time
import os

# --- 1. CONFIGURATION ---
GATEWAY_URL = "https://gw.hodlai.fun"
PRIVATE_KEY = os.getenv("AGENT_KEY", "YOUR_PRIVATE_KEY_HERE") # Replace with your Agent's Private Key

def main():
    print(f"🚀 Connecting to HODLAI Protocol...")
    
    # --- 2. AUTHENTICATION (SIWE) ---
    w3 = Web3()
    account = w3.eth.account.from_key(PRIVATE_KEY)
    print(f"🤖 Agent Identity: {account.address}")

    # Sign the "Proof of Hold" message
    # Format: "Login to HODLAI Protocol\nWallet: {address}\nTimestamp: {current_unix_seconds}"
    timestamp = int(time.time())
    msg_text = f"Login to HODLAI Protocol\nWallet: {account.address}\nTimestamp: {timestamp}"
    signature = account.sign_message(encode_defunct(text=msg_text)).signature.hex()

    # Exchange Signature for Bearer Token
    # This Token is valid for 24 hours. Cache it to avoid re-signing every request.
    auth = requests.post(f"{GATEWAY_URL}/auth/login", json={
        "message": msg_text, "signature": signature
    }).json()

    if 'error' in auth:
        print(f"❌ Auth Failed: {auth['error']}")
        return

    token = auth['token']
    quota = auth['quota']['quota']
    print(f"✅ Logged In! Daily Credits: {quota}")

    if quota <= 0:
        print("⚠️  Credits Depleted. Buy more $HODLAI to recharge instantly.")
        return

    # --- 3. INFERENCE (OpenAI Format) ---
    print("\n🧠 Thinking...")
    headers = { "Authorization": f"Bearer {token}" }
    
    # Example: Using a Premium Model (Cost: 10 Credits/req)
    # Available: gpt-5.2-pro, claude-opus-4.5, gemini-3-pro, etc.
    resp = requests.post(
        f"{GATEWAY_URL}/v1/chat/completions",
        headers=headers,
        json={
            "model": "gpt-5.2-pro", 
            "messages": [{"role": "user", "content": "Explain the concept of 'Holding is Access' in 50 words."}]
        }
    )

    print(f"🤖 Output:\n{resp.json()['choices'][0]['message']['content']}")

if __name__ == "__main__":
    main()

API Reference

POST /auth/login

Exchange an Ethereum Signature for a session token.

# Payload
{ "message": "Login...", "signature": "0x..." }
POST /v1/chat/completions

Standard OpenAI-compatible chat completion. Requires Bearer Token.

Header Value
Authorization Bearer <token>
GET /v1/models

List available models (gpt-4o, o1-preview, deepseek-chat, etc.).

Troubleshooting

402 Payment Required

Your Agent doesn't hold enough $HODLAI or your purchase was less than 1 minute ago (Anti-Flashloan protection).

401 Unauthorized

Your signature is invalid or your session token expired (Tokens last 24h).
